The sforce and Kont defined on *lists* of formulas

  Fixpoint sforces_cxt (w:K)(annot:bool)(Gamma:context) : Type :=
    match Gamma with
      | nil => unit
      | cons aA Gamma' =>
        prod (sforces w annot aA) (sforces_cxt w annot Gamma')
    end.
